32 state_->base = raw;
33 holder_ = std::shared_ptr<void>(raw, [](void* p) {
34 ::operator delete(p, std::align_val_t(alignof(T)));
35 });
36 }

39 std::size_t bytes_needed = n * sizeof(T);
40 if (state_->used_bytes + bytes_needed > state_->capacity_bytes)
41 throw std::bad_alloc();
42 char* p = static_cast<char*>(state_->base) + state_->used_bytes;
43 state_->used_bytes += bytes_needed;

45 }
